model Note {
  id        Int      @id @default(autoincrement())
  user      User     @relation(fields: [userId], references: [id])
  userId    Int
  filename  String
  summary   String?
  createdAt DateTime @default(now())
  quizzes   Quiz[]
}

model Quiz {
  id        Int      @id @default(autoincrement())
  user      User     @relation(fields: [userId], references: [id])
  userId    Int
  note      Note     @relation(fields: [noteId], references: [id])
  noteId    Int
  questions Json
  score     Int?
  takenAt   DateTime?
}

user model update{
    quizzes   Quiz[]
  notes     Note[]
}
